const SHEET_NAME = ""; // 留空 = 使用第一個分頁 const HEADERS = ["ID", "CreatedAt", "Status", "Title", "Priority", "Note"]; function _json(data) { return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); } function _sheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = SHEET_NAME ? ss.getSheetByName(SHEET_NAME) : ss.getSheets()[0]; if (!sheet) throw new Error("No sheet found"); return sheet; } function _headers(sheet) { const firstCell = sheet.getRange(1, 1).getValue(); if (!firstCell) { sheet.getRange(1, 1, 1, HEADERS.length).setValues([HEADERS]); return HEADERS; } const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; const missing = HEADERS.filter(function(name) { return !headers.some(function(h) { return String(h).toLowerCase() === String(name).toLowerCase(); }); }); if (missing.length) throw new Error("Missing header: " + missing.join(", ")); return headers; } function _pick(record, header) { if (record[header] !== undefined) return record[header]; const key = Object.keys(record).find(function(k) { return k.toLowerCase() === String(header).toLowerCase(); }); return key ? record[key] : ""; } function doGet() { try { const sheet = _sheet(); const headers = _headers(sheet); const values = sheet.getDataRange().getValues().slice(1); const records = values .filter(function(row) { return row.some(function(cell) { return cell !== ""; }); }) .map(function(row) { const record = {}; headers.forEach(function(h, i) { record[h] = row[i]; }); return record; }); return _json({ ok: true, records: records }); } catch (err) { return _json({ ok: false, error: String(err.message || err) }); } } function doPost(e) { try { const body = JSON.parse(e.postData.contents || "{}"); if (body.action === "create") return _create(body.record || {}); if (body.action === "update") return _update(body.id, body.changes || {}); return _json({ ok: false, error: "Unknown action" }); } catch (err) { return _json({ ok: false, error: String(err.message || err) }); } } function _create(record) { const sheet = _sheet(); const headers = _headers(sheet); const row = headers.map(function(h) { return _pick(record, h); }); sheet.appendRow(row); return _json({ ok: true, action: "create" }); } function _update(id, changes) { const sheet = _sheet(); const headers = _headers(sheet); const values = sheet.getDataRange().getValues(); const idIndex = headers.findIndex(function(h) { return String(h).toLowerCase() === "id"; }); if (idIndex === -1) { return _json({ ok: false, error: "ID column not found in headers" }); } for (let r = 1; r < values.length; r++) { if (String(values[r][idIndex]) === String(id)) { Object.keys(changes).forEach(function(key) { const header = headers.find(function(h) { return String(h).toLowerCase() === key.toLowerCase(); }); if (header) { sheet.getRange(r + 1, headers.indexOf(header) + 1).setValue(changes[key]); } }); return _json({ ok: true, action: "update" }); } } return _json({ ok: false, error: "Record not found" }); }
